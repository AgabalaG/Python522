Конструктор объектов Person

Создание объектов со свойствами name, age, job и методом who()

function Person(name, age, job) {
    this.name = name;
    this.age = age;
    this.job = job;
    
    this.who = function() {
        return Я ${this.name} мне ${this.age} лет. Я работаю ${this.job}ом.;
    };
}
